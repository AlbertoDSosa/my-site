var gulp = require('gulp')

gulp.task('build:css', function () {
	gulp.src('src/styles/estilos.styl')
	.pipe(stylus({
		use: nib(),
		'include css': true
	}))
	.pipe(gulp.dest(build/main.css));
});

gulp.task('default', ['build:css']);
